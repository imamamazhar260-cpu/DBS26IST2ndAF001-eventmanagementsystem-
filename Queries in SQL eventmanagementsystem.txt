--1. ROLE TABLE
CREATE TABLE Role (
    RoleID INT PRIMARY KEY IDENTITY(1,1),
    RoleName VARCHAR(50) NOT NULL UNIQUE
);


--2. USER TABLE
CREATE TABLE [User] (
    UserID INT PRIMARY KEY IDENTITY(1,1),

    FullName VARCHAR(100) NOT NULL,

    Email VARCHAR(100)
    UNIQUE NOT NULL,

    Password VARCHAR(100)
    NOT NULL,

    Phone VARCHAR(20),

    Gender VARCHAR(10),

    RoleID INT,

    CreatedAt DATETIME
    DEFAULT GETDATE(),

    FOREIGN KEY (RoleID)
    REFERENCES Role(RoleID)
);

--3. ORGANIZER TABLE
CREATE TABLE Organizer (
    OrganizerID INT PRIMARY KEY IDENTITY(1,1),

    UserID INT UNIQUE,

    OrganizationName VARCHAR(100),

    ExperienceYears INT
    CHECK (ExperienceYears >= 0),

    FOREIGN KEY (UserID)
    REFERENCES [User](UserID)
);


--4. CUSTOMER TABLE
CREATE TABLE Customer (
    CustomerID INT PRIMARY KEY IDENTITY(1,1),

    UserID INT UNIQUE,

    Address VARCHAR(200),

    FOREIGN KEY (UserID)
    REFERENCES [User](UserID)
);


--5. STAFF TABLE
CREATE TABLE Staff (
    StaffID INT PRIMARY KEY IDENTITY(1,1),

    UserID INT UNIQUE,

    Position VARCHAR(50),

    Salary DECIMAL(10,2)
    CHECK (Salary >= 0),

    FOREIGN KEY (UserID)
    REFERENCES [User](UserID)
);


--6. VENUE TABLE
CREATE TABLE Venue (
    VenueID INT PRIMARY KEY IDENTITY(1,1),

    VenueName VARCHAR(100)
    NOT NULL,

    Location VARCHAR(200),

    Capacity INT
    CHECK (Capacity > 0),

    ContactNumber VARCHAR(20)
);


--7. CATEGORY TABLE
CREATE TABLE Category (
    CategoryID INT PRIMARY KEY IDENTITY(1,1),

    CategoryName VARCHAR(50)
    NOT NULL UNIQUE
);



--8. EVENT TABLE
CREATE TABLE Event (
    EventID INT PRIMARY KEY IDENTITY(1,1),

    EventTitle VARCHAR(100)
    NOT NULL,

    Description TEXT,

    EventDate DATE
    NOT NULL,

    StartTime TIME,

    EndTime TIME,

    TicketPrice DECIMAL(10,2),

    TotalSeats INT
    CHECK (TotalSeats > 0),

    VenueID INT,

    CategoryID INT,

    OrganizerID INT,

    FOREIGN KEY (VenueID)
    REFERENCES Venue(VenueID),

    FOREIGN KEY (CategoryID)
    REFERENCES Category(CategoryID),

    FOREIGN KEY (OrganizerID)
    REFERENCES Organizer(OrganizerID)
);


--9. EVENTSCHEDULE TABLE
CREATE TABLE EventSchedule (
    ScheduleID INT PRIMARY KEY IDENTITY(1,1),

    EventID INT,

    ActivityName VARCHAR(100),

    ActivityTime TIME,

    FOREIGN KEY (EventID)
    REFERENCES Event(EventID)
);


--10. AVAILABLESLOT TABLE
CREATE TABLE AvailableSlot (
    SlotID INT PRIMARY KEY IDENTITY(1,1),

    EventID INT,

    AvailableSeats INT
    CHECK (AvailableSeats >= 0),

    FOREIGN KEY (EventID)
    REFERENCES Event(EventID)
);



==11. BOOKING TABLE
CREATE TABLE Booking (
    BookingID INT PRIMARY KEY IDENTITY(1,1),

    CustomerID INT,

    EventID INT,

    BookingDate DATETIME
    DEFAULT GETDATE(),

    NumberOfTickets INT
    CHECK (NumberOfTickets > 0),

    BookingStatus VARCHAR(20)
    CHECK (BookingStatus IN
    ('Confirmed','Pending','Cancelled')),

    FOREIGN KEY (CustomerID)
    REFERENCES Customer(CustomerID),

    FOREIGN KEY (EventID)
    REFERENCES Event(EventID)
);



--12. PAYMENT TABLE
CREATE TABLE Payment (
    PaymentID INT PRIMARY KEY IDENTITY(1,1),

    BookingID INT UNIQUE,

    Amount DECIMAL(10,2)
    CHECK (Amount >= 0),

    PaymentDate DATETIME
    DEFAULT GETDATE(),

    PaymentMethod VARCHAR(50),

    PaymentStatus VARCHAR(20)
    CHECK (PaymentStatus IN
    ('Paid','Pending','Refunded')),

    FOREIGN KEY (BookingID)
    REFERENCES Booking(BookingID)
);


--13. TICKET TABLE
CREATE TABLE Ticket (
    TicketID INT PRIMARY KEY IDENTITY(1,1),

    BookingID INT,

    TicketNumber VARCHAR(50)
    UNIQUE NOT NULL,

    SeatNumber VARCHAR(20),

    QRCode VARCHAR(200),

    FOREIGN KEY (BookingID)
    REFERENCES Booking(BookingID)
);


--14. VENDOR TABLE
CREATE TABLE Vendor (
    VendorID INT PRIMARY KEY IDENTITY(1,1),

    VendorName VARCHAR(100)
    NOT NULL,

    ServiceType VARCHAR(50),

    ContactNumber VARCHAR(20)
);


--15. EVENTVENDOR TABLE
CREATE TABLE EventVendor (
    EventVendorID INT PRIMARY KEY IDENTITY(1,1),

    EventID INT,

    VendorID INT,

    FOREIGN KEY (EventID)
    REFERENCES Event(EventID),

    FOREIGN KEY (VendorID)
    REFERENCES Vendor(VendorID)
);


--16. FEEDBACK TABLE
CREATE TABLE Feedback (
    FeedbackID INT PRIMARY KEY IDENTITY(1,1),

    CustomerID INT,

    EventID INT,

    Rating INT
    CHECK (Rating BETWEEN 1 AND 5),

    Comments TEXT,

    FeedbackDate DATETIME
    DEFAULT GETDATE(),

    FOREIGN KEY (CustomerID)
    REFERENCES Customer(CustomerID),

    FOREIGN KEY (EventID)
    REFERENCES Event(EventID)
);


--17. NOTIFICATION TABLE
CREATE TABLE Notification (
    NotificationID INT PRIMARY KEY IDENTITY(1,1),

    UserID INT,

    Message VARCHAR(255),

    NotificationDate DATETIME
    DEFAULT GETDATE(),

    FOREIGN KEY (UserID)
    REFERENCES [User](UserID)
);

--18. LOGS TABLE
CREATE TABLE Logs (
    LogID INT PRIMARY KEY IDENTITY(1,1),

    ErrorMessage VARCHAR(255),

    LogDate DATETIME
    DEFAULT GETDATE(),

    UserID INT,

    FOREIGN KEY (UserID)
    REFERENCES [User](UserID)
);

----views----------


--upcoming event view
CREATE VIEW UpcomingEvents AS
SELECT
EventID,
EventTitle,
EventDate,
TicketPrice
FROM Event
WHERE EventDate >= GETDATE();



--Booking Details View
CREATE VIEW BookingDetails AS
SELECT
B.BookingID,
U.FullName,
E.EventTitle,
B.NumberOfTickets,
B.BookingStatus

FROM Booking B

JOIN Customer C
ON B.CustomerID = C.CustomerID

JOIN [User] U
ON C.UserID = U.UserID

JOIN Event E
ON B.EventID = E.EventID;


--Payment Summary View
CREATE VIEW PaymentSummary AS
SELECT
PaymentID,
Amount,
PaymentMethod,
PaymentStatus
FROM Payment;

--Event Vendor View

CREATE VIEW 
EventVendorDetails AS
SELECT
E.EventTitle,
V.VendorName,
V.ServiceType

FROM EventVendor EV

JOIN Event E
ON EV.EventID = E.EventID

JOIN Vendor V
ON EV.VendorID = V.VendorID;



--Feedback View
CREATE VIEW FeedbackDetails AS
SELECT
F.FeedbackID,
E.EventTitle,
F.Rating,
F.Comments

FROM Feedback F

JOIN Event E
ON F.EventID = E.EventID;

--search Event Procedure
CREATE PROCEDURE SearchEvent
@Title VARCHAR(100)

AS
BEGIN

SELECT *
FROM Event
WHERE EventTitle LIKE '%' + @Title + '%'

END


--delete event procedure
CREATE PROCEDURE DeleteEvent
@EventID INT

AS
BEGIN

DELETE FROM Event
WHERE EventID = @EventID

END
